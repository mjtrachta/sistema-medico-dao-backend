# Configuración para usar Gmail con App Password
# 1. Ir a Google Account settings
# 2. Enable 2-Factor Authentication
# 3. Generate App Password para "Mail"
# 4. Usar ese password aquí

# Configuración Email en .env
EMAIL_ENABLED=true
SMTP_SERVER=smtp.gmail.com
SMTP_PORT=587
SMTP_USERNAME=tu-email@gmail.com
SMTP_PASSWORD=tu-app-password-generado
SMTP_USE_TLS=true
